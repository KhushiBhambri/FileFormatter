import java.io.File;
import java.io.IOException;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;

public class JsonAttributeSearch {
  
  public static void main(String[] args) {
    ObjectMapper mapper = new ObjectMapper();
    File jsonFile = new File("example.json");

    try {
      JsonNode rootNode = mapper.readTree(jsonFile);
      JsonNode attributeNode = rootNode.path("attributeName");

      if (attributeNode.isMissingNode()) {
        System.out.println("Attribute not found in JSON file.");
      } else {
        System.out.println("Attribute value: " + attributeNode.asText());
      }
    } catch (IOException e) {
      e.printStackTrace();
    }
  }
}
